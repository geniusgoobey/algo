  public static void Insertionsort()
        {
            int[] arr = new int[] { 7, 2, 4, 9, 3, 6, 1, 8 };

            for (int i = 0; i < arr.Length; i++)
            {
                int temp = arr[i];
                int j = i - 1;

                while (j >= 0 && arr[j] > temp)
                {
                    arr[j + 1] = arr[j];
                    j--;
                }
                arr[j + 1] = temp;

            }
        }
